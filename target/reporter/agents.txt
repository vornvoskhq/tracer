Agent Registry Design

# agents/registry.py
class AgentRegistry:
    """Central registry for all available agents."""
    
    _agents: Dict[str, Type[BaseAgent]] = {}
    
    @classmethod
    def register(cls, agent_class: Type[BaseAgent]):
        """Register an agent class."""
        cls._agents[agent_class.name] = agent_class
    
    @classmethod
    def get_agent(cls, name: str) -> Type[BaseAgent]:
        """Get an agent class by name."""
        if name not in cls._agents:
            raise ValueError(f"Agent '{name}' not found. Available: {list(cls._agents.keys())}")
        return cls._agents[name]
    
    @classmethod
    def list_agents(cls) -> Dict[str, Dict[str, str]]:
        """List all registered agents with their info."""
        return {
            name: {
                "name": agent_class.name,
                "description": agent_class.description,
                "version": agent_class.version
            }
            for name, agent_class in cls._agents.items()
        }

class StockAnalyst(BaseAgent):
    """Agent for comprehensive stock analysis."""
    
    name = "stock_analyst"
    description = "Analyzes individual stocks using multiple tools and generates comprehensive reports"
    version = "1.0"
    
    def __init__(self, config: Config, model_manager: ModelManager):
        super().__init__(config, model_manager)
        # Register available tools
        self._setup_tools()
    
    def _setup_tools(self):
        """Set up tools available to this agent."""
        self.toolchain.tools.update({
            'stock_identity': StockIdentity(self.model_manager),
            'summarizer': Summarizer(self.model_manager),
            # Future tools could be added here
        })
    
    async def process(self, request: AgentRequest) -> AgentResponse:
        """Process stock analysis request."""
        # Phase 1: Rule-based chain planning
        chain_plan = self._plan_analysis_chain(request)
        
        # Execute the chain
        results = await self.execute_chain(chain_plan)
        
        # Format response
        return self._format_response(results, request)
    
    def _plan_analysis_chain(self, request: AgentRequest) -> ChainPlan:
        """Plan the analysis chain based on request."""
        # Rule-based logic for Phase 1
        steps = [
            ChainStep(tool='stock_identity', params={'symbol': request.symbol}),
            ChainStep(tool='summarizer', params={
                'text': '{step_0.business_summary}',
                'instruction': request.instruction or 'Provide comprehensive analysis'
            })
        ]
        return ChainPlan(steps=steps)
